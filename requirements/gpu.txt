# GPU dependencies — EXPORT / TENSORRT / INT8 / SERVING stage.
# Install ONLY on CUDA-enabled machines (Colab / Lambda / RunPod). Do NOT install
# on Mac. This stack is pinned to transformers <5.0 (the version optimum-onnx /
# the ONNX export expects).
#
# NOTE: AWQ/GPTQ quantization lives in a SEPARATE environment — see
# requirements/gpu-quant.txt. Those backends (llmcompressor / gptqmodel) require
# transformers 5.x, which is incompatible with this export stack, so they must run
# in their own Colab session. The two stages chain via saved checkpoints on disk.

--extra-index-url https://download.pytorch.org/whl/cu121

torch==2.3.0+cu121
transformers>=4.40,<5.0
onnx>=1.16

# Export verification uses ONNX Runtime. Use the CPU build: onnxruntime-gpu wheels
# are pinned to a CUDA major (recent ones pull a cu13 build -> "libcudart.so.13"
# load error on Colab's CUDA 12). The parity check is correctness-only, so CPU ORT
# is sufficient and avoids the mismatch entirely.
onnxruntime>=1.18

# TensorRT: the bare `tensorrt` meta-package resolves to a cu13 build on current
# indexes (-> libcudart.so.13 mismatch on CUDA 12). Pin the cu12 build explicitly.
tensorrt-cu12>=10.0
pycuda>=2024.1
numpy>=1.26,<2.0
pyyaml>=6.0

# Export stack
# optimum 2.x split the ONNX exporter into the `optimum-onnx` package, which
# re-adds optimum.exporters.onnx / optimum.onnxruntime under the optimum namespace.
optimum-onnx>=0.1

# Optimization stack
flash-attn>=2.5
# INT8 quantization (calibration-free LLM.int8()). Lives in this stack because
# bitsandbytes is transformers-version-agnostic; AWQ/GPTQ do not (see gpu-quant.txt).
bitsandbytes>=0.43
pynvml>=11.5
